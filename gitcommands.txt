 2. STATUS & INSPECT
----------------------------------------------------------------

% Path to existing repo (set to repo root)
repoPath = "C:\path\to\your\repo"; % change this
fileName = "CS1.slx";

% Change folder to repo (optional)
oldPwd = pwd;
cd(repoPath);

try
    % Open repository object
    repo = gitrepo;

    % Check if file exists in the filesystem
    if ~isfile(fullfile(repoPath, fileName))
        error("FileNotFound:CS1", "File '%s' not found in repository folder.", fileName);
    end

    % If file already tracked and unmodified, report and exit
    tracked = ismember(fileName, repo.TrackedFiles);
    modified = ismember(fileName, repo.ModifiedFiles);
    untracked = ismember(fileName, repo.UntrackedFiles);

    if tracked && ~modified
        fprintf("'%s' is already tracked and has no changes to commit.\n", fileName);
    else
        % Stage the file
        add(repo, fileName);
        fprintf("Staged '%s'.\n", fileName);

        % Commit with a default message (customize if desired)
        msg = "Add or update CS1.slx";
        commitInfo = commit(repo, Message=msg);
        fprintf("Committed: %s\n", commitInfo.Id);

        % Determine current branch and upstream
        branchName = repo.Branch;
        upstream = "";
        if isprop(repo,"UpstreamBranch") && ~isempty(repo.UpstreamBranch)
            upstream = repo.UpstreamBranch;
        end

        % Push: if upstream exists, push default; otherwise push to origin/<branch>
        try
            if ~isempty(upstream)
                push(repo);
                fprintf("Pushed to upstream '%s'.\n", upstream);
            else
                push(repo, Remote="origin", RemoteBranch=branchName);
                fprintf("Pushed to origin/%s.\n", branchName);
            end
        catch pushErr
            fprintf("Push failed: %s\n", pushErr.message);
            fprintf("You may need to set upstream or authenticate. Try pushing manually.\n");
        end
    end
catch ME
    fprintf("Error: %s\n", ME.message);
end

----------------------------------------------------------------
 3. STAGE & COMMIT
----------------------------------------------------------------

Stage ALL changed files:
  system('git add .')

Stage a specific .m file:
  system('git add filename.m')

Stage a specific Simulink model:
  system('git add PID_model.slx')

Stage all .m files only:
  system('git add *.m')

Stage all .slx files only:
  system('git add *.slx')

Commit with a message:
  system('git commit -m "Your commit message here"')

Example commits for this project:
  system('git commit -m "Add PID controller initial design"')
  system('git commit -m "Update power factor correction model"')
  system('git commit -m "Tune PID gains for better response"')
  system('git commit -m "Add simulation results and plots"')


----------------------------------------------------------------
 4. PUSH & PULL (SYNC WITH GITHUB)
----------------------------------------------------------------

Push to GitHub (main branch):
  system('git push origin main')

Push to GitHub (master branch):
  system('git push origin master')

Pull latest changes from GitHub:
  system('git pull origin main')

Clone the repo into current folder:
  system('git clone https://github.com/devkesav/PID-Control-Design-for-Power-Factor-Correction.git')

Fetch without merging:
  system('git fetch origin')

Push a new branch to GitHub:
  system('git push -u origin branch-name')


----------------------------------------------------------------
 5. BRANCHES
----------------------------------------------------------------

Create a new branch:
  system('git checkout -b branch-name')

Switch to an existing branch:
  system('git checkout branch-name')

Switch back to main:
  system('git checkout main')

Merge a branch into main:
  system('git merge branch-name')

Delete a local branch:
  system('git branch -d branch-name')

Delete a remote branch:
  system('git push origin --delete branch-name')

Suggested branches for this project:
  system('git checkout -b pid-tuning')
  system('git checkout -b simulation-test')
  system('git checkout -b power-factor-analysis')


----------------------------------------------------------------
 6. UNDO & FIX MISTAKES
----------------------------------------------------------------

Unstage a file (keep changes):
  system('git reset HEAD filename.m')

Discard all local changes (CAREFUL - cannot undo):
  system('git checkout -- filename.m')

Undo last commit but keep changes:
  system('git reset --soft HEAD~1')

Revert a commit safely (creates new commit):
  system('git revert HEAD')

Stash uncommitted changes temporarily:
  system('git stash')

Restore stashed changes:
  system('git stash pop')

List all stashes:
  system('git stash list')

Clear all stashes:
  system('git stash clear')

  system('git config --global diff.tool matlab')

 8. DAILY WORKFLOW (COPY & USE)
----------------------------------------------------------------

--- START OF DAY: Get latest from GitHub ---
  system('git pull origin main')

--- DURING WORK: Check what changed ---
  system('git status')

--- SAVE WORK: Stage and commit ---
  system('git add .')
  system('git commit -m "Describe what you changed"')

--- END OF DAY: Push to GitHub ---
  system('git push origin main')

 9. .gitignore — FILES TO EXCLUDE
----------------------------------------------------------------

Create a .gitignore file from MATLAB (run once):

  fid = fopen('.gitignore', 'w');
  fprintf(fid, '# MATLAB autosave\n*.asv\n\n');
  fprintf(fid, '# Simulink autosave & cache\n*.slxc\n*.bak\n\n');
  fprintf(fid, '# Simulink code generation\nslprj/\ncodegen/\n\n');
  fprintf(fid, '# Simulation output files\n*.mat\n*.log\n\n');
  fprintf(fid, '# MATLAB figures\n*.fig\n\n');
  fprintf(fid, '# OS files\n.DS_Store\nThumbs.db\n');
  fclose(fid);
  disp('.gitignore created!')


----------------------------------------------------------------
 10. USEFUL TIPS
----------------------------------------------------------------

- ans = 0 after system() means SUCCESS (no error)
- ans = 1 or higher means there was an ERROR

- Always pull before you push to avoid conflicts:
    system('git pull origin main')
    system('git push origin main')

- Use meaningful commit messages so you can track changes:
    GOOD: "Fix PID integral windup issue"
    BAD : "update" or "fix"

- Never commit large .mat data files — add them to .gitignore

- Your Personal Access Token (PAT) is your GitHub password
  when MATLAB/Git asks for a password during push/pull

- To check your current directory in MATLAB before cloning:
    pwd

- To change directory in MATLAB:
    cd 'C:\Users\YourName\Documents\MATLAB\YourProject'
